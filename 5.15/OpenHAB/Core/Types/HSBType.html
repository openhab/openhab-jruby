def <=>(other)
  logger.trace("(#{self.class}) #{self} <=> #{other} (#{other.class})")
  if other.is_a?(HSBType)
    [brightness, hue, saturation] <=> [other.brightness, other.hue, other.saturation]
  else
    super
  end
end
