def inspect
  parent = " (#{parent_uid})" unless parent_uid.empty?
  "#<OpenHAB::Core::Items::Semantics::#{name}#{parent} " \
    "label=#{label.inspect} " \
    "description=#{description.inspect} " \
    "synonyms=#{synonyms.to_a.inspect}>"
end
